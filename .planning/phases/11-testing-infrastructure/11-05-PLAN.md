        validate_dataframe_schema(df, "test_schema", Path("test.parquet"), strict=False)


    def test_validate_dataframe_schema_value_out_of_range():
        """Test schema validation detects values outside range."""
        df = pd.DataFrame({
            "event_type": [0, 5, 15],  # 15 is outside range [0, 10]
        })
        schema = {
            "required_columns": ["event_type"],
            "column_types": {"event_type": "int"},
            "value_ranges": {"event_type": {"min": 0, "max": 10}},
        }

        with pytest.raises(DataValidationError, match="event_type.*above max"):
            validate_dataframe_schema(df, "test_schema", Path("test.parquet"))

        ("not_a_number", "type mismatch"),
        (-1, "below min"),
        (11, "above max"),

            "value_ranges": {"event_type": {"min": 0, "max": 10}},

            with pytest.raises(DataValidationError, match="type"):
                validate_dataframe_schema(df, "test_schema", Path("test.parquet"))
        elif invalid_value < 0:
            with pytest.raises(DataValidationError, match="below min"):
                validate_dataframe_schema(df, "test_schema", Path("test.parquet"))
        else:
            with pytest.raises(DataValidationError, match="above max"):
                validate_dataframe_schema(df, "test_schema", Path("test.parquet"))

    import pytest
    import os
    from 2_Scripts.shared.env_validation import (
        validate_env_schema,
        EnvValidationError,
    )


    def test_validate_env_schema_invalid_type(monkeypatch):
        """Test environment validation raises error for invalid type."""
        monkeypatch.setenv("TEST_INT", "not_a_number")
        schema = {
            "TEST_INT": {
                "required": False,
                "type": int,
            }
        }

        with pytest.raises(EnvValidationError, match="Invalid type"):
            validate_env_schema(schema)


    def test_validate_env_schema_missing_required(monkeypatch):
        """Test environment validation raises error for missing required var."""
        monkeypatch.delenv("REQUIRED_VAR", raising=False)
        schema = {
            "REQUIRED_VAR": {
                "required": True,
                "type": str,
                "description": "This is required",
            }
        }

        with pytest.raises(EnvValidationError, match="Required environment variable not set"):
            validate_env_schema(schema)


    def test_validate_env_schema_empty_string(monkeypatch):
        """Test environment validation handles empty string values."""
        monkeypatch.setenv("TEST_VAR", "")
        schema = {
            "TEST_VAR": {
                "required": False,
                "type": str,
            }
        }

        # Empty string is valid for str type
        validated = validate_env_schema(schema)
        assert validated["TEST_VAR"] == ""


    def test_validate_env_schema_whitespace_values(monkeypatch):
        """Test environment validation handles whitespace in values."""
        monkeypatch.setenv("TEST_INT", "  42  ")
        monkeypatch.setenv("TEST_BOOL", "  yes  ")
        schema = {
            "TEST_INT": {"required": False, "type": int},
            "TEST_BOOL": {"required": False, "type": bool},
        }

        validated = validate_env_schema(schema)

        # int should strip whitespace
        assert validated["TEST_INT"] == 42

        # bool should strip whitespace
        assert validated["TEST_BOOL"] is True


    @pytest.mark.parametrize("bool_value,expected", [
        ("true", True),
        ("True", True),
        ("TRUE", True),
        ("false", False),
        ("False", False),
        ("FALSE", False),
        ("1", True),
        ("0", False),
        ("yes", True),
        ("no", False),
    ])
    def test_validate_env_schema_bool_variations(bool_value, expected, monkeypatch):
        """Test environment validation handles various bool string formats."""
        monkeypatch.setenv("TEST_BOOL", bool_value)
        schema = {
            "TEST_BOOL": {
                "required": False,
                "type": bool,
            }
        }

        validated = validate_env_schema(schema)
        assert validated["TEST_BOOL"] == expected


    def test_env_validation_error_messages_are_clear(monkeypatch):
        """Test that validation error messages are clear and actionable."""
        monkeypatch.delenv("MISSING_VAR", raising=False)
        schema = {

            assert "required" in error_msg.lower(), "Error should mention requirement"

    ```python
    import pytest
    from pathlib import Path
    from 2_Scripts.shared.subprocess_validation import (
        validate_script_path,
    )


    def test_validate_script_path_path_traversal(tmp_path):
        """Test path validation prevents path traversal attacks."""
        # Create a test directory structure
        allowed_dir = tmp_path / "allowed"
        allowed_dir.mkdir()

        # Test various path traversal attempts
        traversal_paths = [
            Path("../etc/passwd"),
            Path("../../etc/passwd"),
            Path("../../../etc/passwd"),
            Path("..\\..\\..\\windows\\system32"),  # Windows-style traversal
            Path("%systemroot%\\system32"),  # Environment variable expansion
        ]

        for test_path in traversal_paths:
            with pytest.raises(ValueError, match="outside allowed directory|invalid path"):
                validate_script_path(test_path, allowed_dir)


    def test_validate_script_path_symlink_outside_allowed(tmp_path):
        """Test path validation blocks symlinks outside allowed directory."""
        import os

        allowed_dir = tmp_path / "allowed"
        allowed_dir.mkdir()

        outside_dir = tmp_path / "outside"
        outside_dir.mkdir()

        # Create a file outside allowed directory
        outside_file = outside_dir / "script.py"
        outside_file.write_text("# malicious script")

        # Create symlink inside allowed directory pointing outside
        symlink_path = allowed_dir / "malicious_link"
        try:
            symlink_path.symlink_to(outside_file)
        except (OSError, NotImplementedError):
            # Symlinks may not be supported on Windows
            pytest.skip("Symlinks not supported on this system")

        # Should reject symlink pointing outside allowed directory
        with pytest.raises(ValueError, match="outside allowed directory"):
            validate_script_path(symlink_path, allowed_dir)


    def test_validate_script_path_relative_path(tmp_path):
        """Test path validation resolves relative paths correctly."""
        allowed_dir = tmp_path / "allowed"
        allowed_dir.mkdir()

        # Create a test script
        script_path = allowed_dir / "script.py"
        script_path.write_text("# test script")

        # Test relative path
        relative_path = Path("allowed/script.py")

        result = validate_script_path(relative_path, tmp_path)

        # Should return absolute path
        assert result.is_absolute()
        assert result == script_path


    def test_validate_script_path_nonexistent_file(tmp_path):
        """Test path validation handles non-existent files."""
        allowed_dir = tmp_path / "allowed"
        allowed_dir.mkdir()

        nonexistent_file = allowed_dir / "nonexistent.py"

        # Should not raise for non-existent file (validation is for security, not existence)
        result = validate_script_path(nonexistent_file, allowed_dir)

        assert result.is_absolute()
        assert result == nonexistent_file


    @pytest.mark.parametrize("test_path,should_pass", [
        (Path("/etc/passwd"), False),  # System file
        (Path("/var/log"), False),  # System directory
        (Path("C:\\Windows\\System32"), False),  # Windows system directory (if on Windows)
        (Path("script.py"), True),  # Relative path in allowed dir
    ])
    def test_validate_script_path_security_checks(test_path, should_pass, tmp_path):
        """Test path validation performs security checks."""
        allowed_dir = tmp_path / "allowed"
        allowed_dir.mkdir()

        if should_pass:
            # Should pass for relative paths
            result = validate_script_path(test_path, allowed_dir)
            assert result.is_absolute()
        else:
            # Should fail for absolute paths outside allowed directory
            with pytest.raises(ValueError, match="outside allowed directory"):
                validate_script_path(test_path, allowed_dir)


    def test_subprocess_validation_error_messages_are_clear(tmp_path):
        """Test that validation error messages are clear and actionable."""
        allowed_dir = tmp_path / "allowed"
        allowed_dir.mkdir()

        malicious_path = Path("/etc/passwd")

        try:
            validate_script_path(malicious_path, allowed_dir)
            pytest.fail("Expected ValueError")
        except ValueError as e:
            error_msg = str(e)

            # Verify error message contains helpful information
            assert malicious_path.name in error_msg or str(malicious_path) in error_msg, (
                "Error should mention the invalid path"
            )
            assert "outside allowed directory" in error_msg.lower() or "invalid" in error_msg.lower(), (
                "Error should explain why path is invalid"
            )
    ```
