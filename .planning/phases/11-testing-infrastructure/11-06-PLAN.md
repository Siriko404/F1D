    ```python
    import pytest
    import pandas as pd
    from pathlib import Path
    import pyarrow.parquet as pq
    from 2_Scripts.shared.chunked_reader import read_in_chunks
    from 2_Scripts.shared.data_validation import validate_dataframe_schema


    # ========== Chunked Reader Edge Cases ==========

    def test_chunked_reader_empty_dataframe(tmp_path):
        """Test chunked reading with empty DataFrame."""
        # Arrange - Create empty Parquet file
        empty_df = pd.DataFrame()
        empty_file = tmp_path / "empty.parquet"
        empty_df.to_parquet(empty_file)

        # Act - Read in chunks
        chunks = list(read_in_chunks(empty_file))

        # Assert - Should return empty list or list with empty DataFrame
        assert len(chunks) == 0 or all(len(chunk) == 0 for chunk in chunks), (
            "Empty file should return empty chunks"
        )


    def test_chunked_reader_single_row(tmp_path):
        """Test chunked reading with single row."""
        # Arrange - Create single-row Parquet file
        single_row_df = pd.DataFrame({"col1": [1], "col2": ["test"]})
        single_row_file = tmp_path / "single_row.parquet"
        single_row_df.to_parquet(single_row_file)

        # Act - Read in chunks
        chunks = list(read_in_chunks(single_row_file))

        # Assert - Should handle single row correctly
        assert len(chunks) > 0, "Should return at least one chunk"
        combined = pd.concat(chunks, ignore_index=True)
        assert len(combined) == 1, "Should preserve single row"
        assert combined.equals(single_row_df), "Should match original DataFrame"


    def test_chunked_reader_all_null_columns(tmp_path):
        """Test chunked reading with all-null columns."""
        # Arrange - Create DataFrame with all-null columns
        all_null_df = pd.DataFrame({
            "col1": [None, None, None],
            "col2": [None, None, None],
        })
        all_null_file = tmp_path / "all_null.parquet"
        all_null_df.to_parquet(all_null_file)

        # Act - Read in chunks
        chunks = list(read_in_chunks(all_null_file))

        # Assert - Should handle all-null columns correctly
        assert len(chunks) > 0, "Should return at least one chunk"
        combined = pd.concat(chunks, ignore_index=True)
        assert combined.equals(all_null_df), "Should match original DataFrame"


    @pytest.mark.parametrize("chunk_size,num_chunks_expected", [
        (1, 3),  # 3 rows, chunk_size 1 -> 3 chunks
        (2, 2),  # 3 rows, chunk_size 2 -> 2 chunks
        (10, 1),  # 3 rows, chunk_size 10 -> 1 chunk
    ])
    def test_chunked_reader_chunk_sizes(tmp_path, chunk_size, num_chunks_expected):
        """Test chunked reading with various chunk sizes."""
        # Arrange - Create 3-row Parquet file
        df = pd.DataFrame({"col1": [1, 2, 3], "col2": ["a", "b", "c"]})
        file_path = tmp_path / "test.parquet"
        df.to_parquet(file_path)

        # Act - Read in chunks with specified size
        chunks = list(read_in_chunks(file_path, chunk_size=chunk_size))

        # Assert - Should return expected number of chunks
        assert len(chunks) == num_chunks_expected, (
            f"Expected {num_chunks_expected} chunks with chunk_size={chunk_size}"
        )

        # Verify combined result matches original
        combined = pd.concat(chunks, ignore_index=True)
        assert combined.equals(df), "Combined chunks should match original DataFrame"


    # ========== Data Validation Edge Cases ==========

    def test_data_validation_empty_dataframe():
        """Test schema validation with empty DataFrame."""
        df = pd.DataFrame()
        schema = {
            "required_columns": [],
            "column_types": {},
        }

        # Should not raise for empty DataFrame with no requirements
        validate_dataframe_schema(df, "test_schema", Path("test.parquet"), strict=False)


    def test_data_validation_single_row():
        """Test schema validation with single-row DataFrame."""
        df = pd.DataFrame({"col1": [1], "col2": ["test"]})
        schema = {
            "required_columns": ["col1", "col2"],
            "column_types": {"col1": "int", "col2": "object"},
        }

        # Should validate single row successfully
        validate_dataframe_schema(df, "test_schema", Path("test.parquet"))


    def test_data_validation_all_null_columns():
        """Test schema validation with all-null columns."""
        df = pd.DataFrame({
            "col1": [None, None, None],
            "col2": [None, None, None],
        })
        schema = {
            "required_columns": ["col1", "col2"],
            "column_types": {"col1": "float", "col2": "float"},
        }

        # Should not raise for all-null columns (null is valid float)
        validate_dataframe_schema(df, "test_schema", Path("test.parquet"))


    # ========== Duplicate Keys Handling ==========

    @pytest.mark.parametrize("merge_type,expected_behavior", [
        ("inner", "should only include matching keys"),
        ("left", "should include all left keys, NaN for unmatched right"),
        ("right", "should include all right keys, NaN for unmatched left"),
        ("outer", "should include all keys from both sides"),
    ])
    def test_duplicate_keys_handling(merge_type, expected_behavior):
        """Test handling of duplicate keys in merges."""
        # Arrange - Create DataFrames with duplicate keys
        left_df = pd.DataFrame({
            "key": [1, 1, 2],
            "value_left": ["a", "b", "c"],
        })

        right_df = pd.DataFrame({
            "key": [1, 1, 3],
            "value_right": ["x", "y", "z"],
        })

        # Act - Perform merge
        result = pd.merge(left_df, right_df, on="key", how=merge_type)

        # Assert - Verify merge behavior
        assert len(result) > 0, "Merge should return non-empty result"
        assert "key" in result.columns, "Result should include key column"

        if merge_type == "inner":
            # Inner merge should only include matching keys (1, 1)
            assert len(result) >= 2, "Inner merge should include duplicate key matches"
        elif merge_type == "outer":
            # Outer merge should include all keys (1, 1, 2, 3)
            assert len(result) >= 4, "Outer merge should include all keys"


    # ========== Missing File Handling ==========

    def test_missing_file_handling(tmp_path):
        """Test handling of missing input files."""
        # Arrange - Path to non-existent file
        missing_file = tmp_path / "nonexistent.parquet"

        # Act & Assert - Should raise FileNotFoundError
        with pytest.raises(FileNotFoundError):
            pd.read_parquet(missing_file)

        with pytest.raises(pq.lib.ArrowInvalid):
            pq.ParquetFile(missing_file)


    # ========== Boundary Values ==========

    @pytest.mark.parametrize("value,should_pass", [
        (0, True),    # Zero boundary
        (1, True),    # Minimum positive
        (100, True),   # Normal value
        (-100, True),  # Negative value
        (3.14, True), # Float value
        (None, True),  # Null value
    ])
    def test_boundary_values_numeric(value, should_pass):
        """Test boundary values for numeric columns."""
        df = pd.DataFrame({"col1": [value]})
        schema = {
            "required_columns": ["col1"],
            "column_types": {"col1": "float"},
        }

        # Should validate boundary values correctly
        if should_pass:
            validate_dataframe_schema(df, "test_schema", Path("test.parquet"))


    @pytest.mark.parametrize("value_range", [
        (-1, 10),    # Range including negative
        (0, 100),    # Standard range
        (0, 1),      # Range [0, 1]
        (0, 1000),   # Large range
    ])
    def test_value_range_validation(value_range):
        """Test value range validation with various ranges."""
        min_val, max_val = value_range

        # Test value within range (should pass)
        df_valid = pd.DataFrame({"col1": [min_val + (max_val - min_val) / 2]})
        schema = {
            "required_columns": ["col1"],
            "column_types": {"col1": "float"},
            "value_ranges": {"col1": {"min": min_val, "max": max_val}},
        }
        validate_dataframe_schema(df_valid, "test_schema", Path("test.parquet"))

        # Test value below range (should fail)
        df_below = pd.DataFrame({"col1": [min_val - 1]})
        with pytest.raises(Exception):  # DataValidationError
            validate_dataframe_schema(df_below, "test_schema", Path("test.parquet"))

        # Test value above range (should fail)
        df_above = pd.DataFrame({"col1": [max_val + 1]})
        with pytest.raises(Exception):  # DataValidationError
            validate_dataframe_schema(df_above, "test_schema", Path("test.parquet"))


    # ========== Type Extremes ==========

    @pytest.mark.parametrize("dtype,test_values", [
        ("int64", [-(2**63), 2**63 - 1]),  # Min/max for int64
        ("int32", [-(2**31), 2**31 - 1]),  # Min/max for int32
        ("float64", [float('-inf'), float('inf')]),  # Min/max for float64
        ("object", ["", "x" * 1000]),  # Empty and long strings
    ])
    def test_type_extremes(dtype, test_values):
        """Test handling of extreme values for different types."""
        for test_value in test_values:
            df = pd.DataFrame({"col1": [test_value]})
            df = df.astype({"col1": dtype})

            # Should handle extreme values without error
            assert len(df) == 1, f"Should handle {dtype} value: {test_value}"
            assert df["col1"].iloc[0] == test_value or (
                pd.isna(df["col1"].iloc[0]) and pd.isna(test_value)
            ), f"Value should be preserved: {test_value}"


    # ========== Combination Edge Cases ==========

    @pytest.mark.parametrize("edge_case,description", [
        ("empty_df", "Empty DataFrame"),
        ("single_row", "Single row DataFrame"),
        ("all_null", "All-null columns"),
        ("duplicate_keys", "Duplicate keys in merge"),
        ("boundary_min", "Minimum boundary value"),
        ("boundary_max", "Maximum boundary value"),
    ])
    def test_chunked_reader_edge_cases_summary(edge_case, description, tmp_path):
        """Test summary of edge cases for chunked reader."""
        if edge_case == "empty_df":
            df = pd.DataFrame()
        elif edge_case == "single_row":
            df = pd.DataFrame({"col1": [1]})
        elif edge_case == "all_null":
            df = pd.DataFrame({"col1": [None, None, None]})
        elif edge_case == "boundary_min":
            df = pd.DataFrame({"col1": [0]})
        elif edge_case == "boundary_max":
            df = pd.DataFrame({"col1": [100]})
        else:
            pytest.skip(f"Edge case not implemented: {edge_case}")

        # Save to Parquet and read in chunks
        if len(df) > 0:
            file_path = tmp_path / "test.parquet"
            df.to_parquet(file_path)
            chunks = list(read_in_chunks(file_path))
            combined = pd.concat(chunks, ignore_index=True)
            assert combined.equals(df), f"{description}: Should match original DataFrame"
        else:
            pytest.skip(f"{description}: Empty DataFrame test")
    ```
