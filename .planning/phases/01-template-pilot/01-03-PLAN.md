3. **Check required fields exist (Python script):**
```python
import json

with open('4_Outputs/1.1_CleanMetadata/latest/stats.json') as f:
    stats = json.load(f)

# Required top-level keys
required_keys = ['step_id', 'timestamp', 'input', 'processing', 'output', 'timing', 'missing_values']
for key in required_keys:
    assert key in stats, f"Missing required key: {key}"
    print(f"[PASS] {key} present")

# Required input fields
input_fields = ['files', 'checksums', 'total_rows', 'total_columns']
for field in input_fields:
    assert field in stats['input'], f"Missing input.{field}"
    print(f"[PASS] input.{field} present")

# Required output fields  
output_fields = ['final_rows', 'final_columns', 'files']
for field in output_fields:
    assert field in stats['output'], f"Missing output.{field}"
    print(f"[PASS] output.{field} present")

# Required timing fields
timing_fields = ['start_iso', 'end_iso', 'duration_seconds']
for field in timing_fields:
    assert field in stats['timing'], f"Missing timing.{field}"
    print(f"[PASS] timing.{field} present")

# Validate checksums not empty
assert len(stats['input']['checksums']) > 0, "No checksums recorded"
print(f"[PASS] Checksums recorded: {len(stats['input']['checksums'])} files")

# Validate row counts are integers > 0
assert stats['input']['total_rows'] > 0, "Input rows should be > 0"
assert stats['output']['final_rows'] > 0, "Output rows should be > 0"
print(f"[PASS] Row counts valid: {stats['input']['total_rows']} -> {stats['output']['final_rows']}")

# Validate timing
assert stats['timing']['duration_seconds'] > 0, "Duration should be > 0"
print(f"[PASS] Duration recorded: {stats['timing']['duration_seconds']}s")

print("\n=== ALL VALIDATION CHECKS PASSED ===")
```
