    ```python
    import re

    def extract_runtime(log_file):
        """Extract runtime from log file."""
        with open(log_file) as f:
            log = f.read()
        # Look for stats.json timing or log time pattern
        match = re.search(r"Total time: ([\d.]+) seconds", log)
        if match:
            return float(match.group(1))
        # Fallback: try parsing "start_time" and "end_time"
        start_match = re.search(r'"start_time": "([^"]+)"', log)
        end_match = re.search(r'"end_time": "([^"]+)"', log)
        if start_match and end_match:
            from datetime import datetime
            start = datetime.fromisoformat(start_match.group(1))
            end = datetime.fromisoformat(end_match.group(1))
            return (end - start).total_seconds()
        return None

    runtime_cold = extract_runtime("run1.log")
    runtime_warm = extract_runtime("run2.log")

    if runtime_cold and runtime_warm:
        speedup = runtime_cold / runtime_warm
        print(f"Cold cache: {runtime_cold:.2f}s")
        print(f"Warm cache: {runtime_warm:.2f}s")
        print(f"Speedup: {speedup:.2f}x")
    ```
