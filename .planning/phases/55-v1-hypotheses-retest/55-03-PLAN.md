```python
def calculate_amihud_illiquidity(df):
    """
    Calculate Amihud (2002) illiquidity measure at firm-year level.

    ILLIQ_{i,y} = (1/D_{i,y}) * sum(|r_{i,y,d}| / VOLD_{i,y,d})

    Where:
        D = number of valid trading days in year
        r = daily return
        VOLD = daily dollar volume

    Returns:
        DataFrame with permno, year, amihud_illiquidity
    """
    # Filter valid observations
    df = df[(df['ret'].notna()) &
            (df['ret'].between(-66, 66)) &
            (df['vol'] > 0) &
            (df['price'] > 0)]

    # Calculate dollar volume (CRSP: vol * price * 100)
    df['dollar_volume'] = df['vol'] * df['price'] * 100

    # Daily Amihud ratio
    df['daily_illiq'] = df['ret'].abs() / df['dollar_volume']

    # Aggregate to firm-year level
    illiq = (df.groupby(['permno', 'year'])
             .agg(amihud_illiquity=('daily_illiq', 'mean'),
                  trading_days=('daily_illiq', 'count'))
             .reset_index())

    # Require minimum trading days
    illiq = illiq[illiq['trading_days'] >= 50]

    # Scale by 1e6
    illiq['amihud_illiquidity'] = illiq['amihud_illiquity'] * 1e6

    return illiq[['permno', 'year', 'amihud_illiquidity']]
```

def calculate_roll_spread(df):
    """
    Calculate Roll (1984) implied spread from price autocovariance.

    SPRD = 2 * sqrt(-cov(r_t, r_{t-1}))

    Returns DataFrame with permno, year, roll_spread
    """
    # Group by permno-year
    def roll_for_group(group):
        if len(group) < 50:  # Need minimum observations
            return pd.Series({'roll_spread': np.nan})
        # Calculate autocovariance
        autocov = group['ret'].autocov(lag=1)
        if autocov >= 0:  # Spread undefined if autocov >= 0
            return pd.Series({'roll_spread': np.nan})
        return pd.Series({'roll_spread': 2 * np.sqrt(-autocov)})

    roll = (df.groupby(['permno', 'year'])
            .apply(roll_for_group)
            .reset_index())
    return roll
```

def winsorize_series(series, lower=0.01, upper=0.99):
    """Winsorize series at specified percentiles"""
    lower_bound = series.quantile(lower)
    upper_bound = series.quantile(upper)
    return series.clip(lower=lower_bound, upper=upper_bound)
```
