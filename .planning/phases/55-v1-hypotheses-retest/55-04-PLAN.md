```python
def create_forward_dv(df, dv_col='amihud_illiquidity'):
    """
    Create forward-looking dependent variable.

    Illiquidity_{t+1} is the illiquidity measure in the NEXT year.
    This allows for causal interpretation: speech_t affects illiquidity_{t+1}.
    """
    df = df.sort_index()
    df['illiquidity_fwd'] = df.groupby(level='gvkey')[dv_col].shift(-1)
    # Remove rows where DV is missing (firm disappears)
    df = df[df['illiquidity_fwd'].notna()]
    return df
```

```python
def run_h7_regression(df, uncertainty_var, dv_col='illiquidity_fwd',
                     controls=None, spec_name='primary'):
    """
    Run single H7 regression.

    Specification:
        Illiquidity_{t+1} = beta0 + beta1*Uncertainty + Controls + FE

    Args:
        df: Panel data with MultiIndex (gvkey, year)
        uncertainty_var: Name of uncertainty IV
        dv_col: Name of dependent variable
        controls: List of control variable names
        spec_name: Specification name for reporting

    Returns:
        dict with coefficients, SEs, p-values, diagnostics
    """
    # Prepare formula variables
    iv_vars = [uncertainty_var]
    if controls:
        iv_vars.extend(controls)

    # Drop rows with missing IV or controls
    model_df = df[[dv_col] + iv_vars].dropna()

    # Set index for PanelOLS
    model_df = model_df.sort_index()

    # Run regression
    from linearmodels.panel import PanelOLS

    model = PanelOLS(
        model_df[dv_col],
        model_df[iv_vars],
        entity_effects=True,  # Firm FE
        time_effects=True     # Year FE
    )

    results = model.fit(
        cov_type='clustered',
        cluster_entity=True  # Firm-clustered SE
    )

    # Extract results
    coef = results.params[uncertainty_var]
    se = results.std_errors[uncertainty_var]
    p_two = results.pvalues[uncertainty_var]
    p_one = p_two / 2 if coef > 0 else 1 - p_two / 2  # H7a: beta > 0

    return {
        'uncertainty_var': uncertainty_var,
        'spec': spec_name,
        'coefficient': coef,
        'se': se,
        'p_two_sided': p_two,
        'p_one_sided': p_one,
        'n': results.nobs,
        'n_firms': model_df.index.get_level_values('gvkey').nunique(),
        'r2_within': results.rsquared_within,
        'results': results
    }
```

```python
from statsmodels.stats.multitest import multipletests

def apply_fdr_regression(results_list, alpha=0.05):
    """
    Apply Benjamini-Hochberg FDR correction to p-values.

    Only applies to primary specification results.
    """
    # Extract p-values from primary spec results
    primary_results = [r for r in results_list if r['spec'] == 'primary']
    p_values = [r['p_one_sided'] for r in primary_results]

    # Apply FDR
    reject, p_corrected, _, _ = multipletests(p_values, alpha=alpha, method='fdr_bh')

    # Update results
    for i, r in enumerate(primary_results):
        r['p_fdr'] = p_corrected[i]
        r['fdr_significant'] = reject[i]

    return results_list
```
