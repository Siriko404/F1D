```python
def create_takeover_timing_variants(df, dv_base='takeover_completed', lags=[0, -1, -2]):
    """
    Create DV variants for different timing assumptions (binary).

    For takeover prediction, we test whether uncertainty predicts
    takeover in same year (t), next year (t+1), or two years ahead (t+2).
    """
    from 3.8_H8TakeoverVariables import create_forward_takeover  # Import if available
    timing_variants = {}
    timing_names = {0: 'concurrent', -1: 'forward', -2: 'lead'}

    for lag in lags:
        df_lag = df.copy()
        if lag == 0:
            df_lag['takeover_timing'] = df_lag[dv_base]
        else:
            df_lag['takeover_timing'] = df_lag.groupby(level='gvkey')[dv_base].shift(-abs(lag))
            df_lag['takeover_timing'] = df_lag['takeover_timing'].fillna(0).astype(int)
        timing_variants[timing_names[lag]] = (df_lag, 'takeover_timing')

    return timing_variants
```

```python
def run_h8_cox_ph(df, uncertainty_var, dv_col='time_to_takeover', controls=None):
    """
    Run Cox proportional hazards model for takeover prediction.

    Uses survival analysis framework: time until takeover event.
    Censoring: firms not acquired by end of sample.

    Returns dict with hazard ratios and statistics.
    """
    try:
        from lifelines import CoxPHFitter
    except ImportError:
        return None  # Skip if lifelines not available

    # Prepare data
    model_df = df[[dv_col, 'takeover_event', uncertainty_var]]
    if controls:
        model_df = model_df.join(df[controls])

    model_df = model_df.dropna()

    # Fit Cox model
    cph = CoxPHFitter()
    cph.fit(model_df, duration_col=dv_col, event_col='takeover_event')

    # Extract results
    summary = cph.summary.loc[uncertainty_var]

    return {
        'model_type': 'cox_ph',
        'uncertainty_var': uncertainty_var,
        'hazard_ratio': summary['exp(coef)'],
        'se': summary['se(coef)'],
        'p_two_sided': summary['p'],
        'ci_lower': summary['exp(coef) lower 95%'],
        'ci_upper': summary['exp(coef) upper 95%'],
    }
```

```python
def run_h8_robustness_suite(df, uncertainty_measures, control_vars):
    """
    Run full robustness suite for H8.

    Dimensions:
    1. Alternative DVs (announced, hostile)
    2. Alternative specs (pooled)
    3. Alternative IVs (CEO-only, Presentation-only)
    4. Timing variants (concurrent, forward, lead)
    5. Alternative models (Cox PH if available)

    Returns list of robustness result dicts.
    """
    results = []

    # Dimension 1: Alternative DVs
    for dv_col, dv_label in H8_ROBUSTNESS_CONFIG['alternative_dvs'].items():
        if dv_col not in df.columns:
            continue
        for uv in uncertainty_measures:
            if uv not in df.columns:
                continue
            result = run_h8_logit(
                df, uv, dv_col, control_vars,
                include_fe=True, spec='primary', dv_type=dv_label
            )
            if result:
                result['dv_type'] = dv_label
                results.append(result)

    # Dimension 2: Alternative specs
    for spec_name, spec_config in H8_ROBUSTNESS_CONFIG['alternative_specs'].items():
        for uv in uncertainty_measures:
            if uv not in df.columns:
                continue
            result = run_h8_logit(
                df, uv, 'takeover_fwd', control_vars,
                include_fe=spec_config['include_fe'],
                spec=spec_name, dv_type='Completed deals'
            )
            if result:
                result['spec'] = spec_name
                results.append(result)

    # Dimension 3: Alternative IVs
    for iv_type, ivs in H8_ROBUSTNESS_CONFIG['alternative_ivs'].items():
        for uv in ivs:
            if uv not in df.columns:
                continue
            result = run_h8_logit(
                df, uv, 'takeover_fwd', control_vars,
                include_fe=True, spec='primary', dv_type='Completed deals',
                iv_type=iv_type
            )
            if result:
                result['iv_type'] = iv_type
                results.append(result)

    # Dimension 4: Timing variants
    timing_variants = create_takeover_timing_variants(df, 'takeover_completed', [0, -1, -2])
    for timing_name, (df_t, dv_col) in timing_variants.items():
        for uv in uncertainty_measures:
            if uv not in df_t.columns:
                continue
            result = run_h8_logit(
                df_t, uv, dv_col, control_vars,
                include_fe=True, spec='primary', dv_type='Completed deals',
                timing=timing_name
            )
            if result:
                result['timing'] = timing_name
                results.append(result)

    # Dimension 5: Cox PH (if available)
    if H8_ROBUSTNESS_CONFIG['alternative_models']['cox_ph']:
        # Try to run Cox PH for primary DV
        cox_result = run_h8_cox_ph(df, uncertainty_measures[0], 'time_to_takeover', control_vars)
        if cox_result:
            results.append(cox_result)

    return results
```

```python
def generate_h8_results_report(primary_results, robustness_results, sample_stats):
    """
    Generate comprehensive H8 results report with robustness.
    """
    # ... existing primary results section ...

    # Add robustness section
    report += "\n## Robustness Suite Results\n\n"

    # Alternative DVs
    report += "### Alternative Dependent Variables\n\n"
    alt_dv_results = [r for r in robustness_results if r.get('dv_type') != 'Completed deals']
    if alt_dv_results:
        report += "| DV | UV | Odds Ratio | 95% CI OR | p (one-tailed) | Significant? |\n"
        report += "|----|----|------------|-----------|----------------|--------------|\n"
        for r in alt_dv_results:
            sig = "Yes" if r['p_one_sided'] < 0.05 else "No"
            or_ci = f"[{r['or_ci_lower']:.2f}, {r['or_ci_upper']:.2f}]"
            dv_type = r.get('dv_type', 'Unknown')
            report += f"| {dv_type} | {r['uncertainty_var']} | {r['odds_ratio']:.2f} | {or_ci} | {r['p_one_sided']:.4f} | {sig} |\n"
    else:
        report += "No alternative DVs available in data.\n"

    # Alternative IVs
    report += "\n### Alternative Independent Variables\n\n"
    for iv_type in ['ceo_only', 'presentation_only', 'qa_only']:
        iv_results = [r for r in robustness_results if r.get('iv_type') == iv_type]
        if iv_results:
            n_sig = sum(1 for r in iv_results if r['p_one_sided'] < 0.05)
            report += f"- **{iv_type}**: {n_sig}/{len(iv_results)} significant\n"

    # Timing tests
    report += "\n### Timing Tests\n\n"
    for timing in ['concurrent', 'forward', 'lead']:
        timing_results = [r for r in robustness_results if r.get('timing') == timing]
        if timing_results:
            n_sig = sum(1 for r in timing_results if r['p_one_sided'] < 0.05)
            report += f"- **{timing}**: {n_sig}/{len(timing_results)} significant\n"

    # Cox PH results (if any)
    cox_results = [r for r in robustness_results if r.get('model_type') == 'cox_ph']
    if cox_results:
        report += "\n### Cox Proportional Hazards Model\n\n"
        report += "Survival analysis results:\n"
        for r in cox_results:
            report += f"- Hazard Ratio: {r['hazard_ratio']:.2f} ({r['ci_lower']:.2f}, {r['ci_upper']:.2f})\n"
            report += f"  p-value: {r['p_two_sided']:.4f}\n"

    # Robustness conclusion
    report += "\n### Robustness Conclusion\n\n"
    n_total = len([r for r in robustness_results if r.get('spec') in ['primary', None]])
    n_sig = sum(1 for r in robustness_results if r['p_one_sided'] < 0.05)
    report += f"Across {n_total} robustness tests, {n_sig} ({n_sig/n_total*100:.1f}%) show significant effects.\n\n"
    if n_sig / n_total > 0.5:
        report += "**Robustness Assessment:** Results are ROBUST - majority of tests significant.\n"
    elif n_sig / n_total > 0.2:
        report += "**Robustness Assessment:** Results are MIXED - some specifications significant.\n"
    else:
        report += "**Robustness Assessment:** Results are NOT ROBUST - few specifications significant.\n"

    return report
```
