```python
def load_sdc_data(sdc_path, year_start, year_end):
    """
    Load and process SDC Platinum M&A data.

    Creates binary takeover indicator at firm-year level.

    Args:
        sdc_path: Path to SDC parquet file
        year_start, year_end: Sample period

    Returns:
        DataFrame with gvkey, year, takeover indicators
    """
    # Load SDC data
    df = pd.read_parquet(sdc_path)

    # Key SDC fields (verify actual column names):
    # - Target GVKEY
    # - Announcement date
    # - Completion status
    # - Deal attitude (hostile/friendly)
    # - Deal type (merger, acquisition)

    # Map SDC columns (adjust based on actual data)
    required_cols = ['target_gvkey', 'announcedate', 'statuscd', 'dealtype']
    df = df[required_cols].dropna(subset=['target_gvkey', 'announcedate'])

    # Convert date and extract year
    df['announcedate'] = pd.to_datetime(df['announcedate'])
    df['year'] = df['announcedate'].dt.year

    # Filter to sample period
    df = df[df['year'].between(year_start, year_end)]

    # Create takeover indicators
    # Primary: Completed deals
    df['takeover_completed'] = (df['statuscd'] == 'C').astype(int)

    # Robustness 1: All announced deals
    df['takeover_announced'] = 1  # All rows are announced by definition

    # Robustness 2: Hostile/unsolicited deals
    hostile_indicators = ['H', 'U', 'R']  # Adjust based on SDC coding
    df['takeover_hostile'] = df['dealtype'].isin(hostile_indicators).astype(int)

    # Aggregate to firm-year level (1 if any takeover in year)
    takeover = df.groupby(['target_gvkey', 'year']).agg({
        'takeover_completed': 'max',
        'takeover_announced': 'max',
        'takeover_hostile': 'max',
    }).reset_index()

    takeover = takeover.rename(columns={'target_gvkey': 'gvkey'})

    return takeover
```

2. Add function to create forward-looking takeover indicator:

```python
def create_forward_takeover(takeover_df):
    """
    Create forward-looking takeover indicator.

    Takeover_{t+1} indicates whether firm becomes target in NEXT year.
    This allows for causal interpretation: uncertainty_t affects takeover_{t+1}.
    """
    takeover_df = takeover_df.sort_values(['gvkey', 'year'])
    takeover_df['takeover_fwd'] = takeover_df.groupby('gvkey')['takeover_completed'].shift(-1)
    takeover_df['takeover_fwd'] = takeover_df['takeover_fwd'].fillna(0).astype(int)
    return takeover_df
```

```python
def merge_h8_data(takeover_df, uncertainty_df, controls_df, manifest_df):
    """
    Merge takeover data with uncertainty and controls.

    Returns analysis-ready dataset.
    """
    # Start with takeover data
    df = takeover_df.copy()

    # Merge uncertainty measures (on gvkey-year)
    df = df.merge(uncertainty_df, on=['gvkey', 'year'], how='left')

    # Merge controls (on gvkey-year)
    df = df.merge(controls_df, on=['gvkey', 'year'], how='left')

    # Filter to sample firms (from manifest)
    sample_firms = manifest_df['gvkey'].unique()
    df = df[df['gvkey'].isin(sample_firms)]

    return df
```

```python
def compute_h8_stats(df):
    """
    Compute descriptive statistics for H8 dataset.

    Special handling for binary takeover indicator.
    """
    stats = {
        'n_obs': len(df),
        'n_firms': df['gvkey'].nunique(),
        'n_years': df['year'].nunique(),
        'year_range': (df['year'].min(), df['year'].max()),
    }

    # Takeover rate (DV)
    stats['takeover_rate'] = df['takeover_fwd'].mean()
    stats['n_takeovers'] = df['takeover_fwd'].sum()

    # Uncertainty measures
    for uv in UNCERTAINTY_MEASURES:
        if uv in df.columns:
            stats[uv] = {
                'mean': df[uv].mean(),
                'std': df[uv].std(),
                'missing': df[uv].isna().sum(),
            }

    # Controls
    for cv in MNA_CONTROL_VARS:
        if cv in df.columns:
            stats[cv] = {
                'mean': df[cv].mean(),
                'std': df[cv].std(),
                'missing': df[cv].isna().sum(),
            }

    return stats
```
