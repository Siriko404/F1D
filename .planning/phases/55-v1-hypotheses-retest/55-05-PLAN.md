def create_alternative_dvs(df):
    """
    Create alternative dependent variables for robustness.

    Returns df with additional DV columns.
    """
    # Roll spread (should exist from variable construction)
    if 'roll_spread' in df.columns:
        df['roll_spread_fwd'] = df.groupby(level='gvkey')['roll_spread'].shift(-1)

    # Bid-ask spread (if available)
    if 'bidask_spread' in df.columns:
        df['bidask_spread_fwd'] = df.groupby(level='gvkey')['bidask_spread'].shift(-1)

    return df

def create_timing_variants(df, dv_base='amihud_illiquidity', lags=[0, -1, -2]):
    """
    Create DV variants for different timing assumptions.

    Args:
        df: Panel data
        dv_base: Base DV column name
        lags: List of lag values (0 = concurrent, -1 = forward, -2 = lead)

    Returns:
        dict with timing_name -> (df, dv_col)
    """
    timing_variants = {}
    timing_names = {0: 'concurrent', -1: 'forward', -2: 'lead'}

    for lag in lags:
        df_lag = df.copy()
        dv_name = f'{dv_base}_lag{abs(lag)}' if lag != 0 else dv_base
        df_lag[dv_name] = df_lag.groupby(level='gvkey')[dv_base].shift(lag)
        df_lag = df_lag[df_lag[dv_name].notna()]  # Remove missing DV
        timing_variants[timing_names[lag]] = (df_lag, dv_name)

    return timing_variants

def run_robustness_suite(df, uncertainty_measures, control_vars):
    """
    Run full robustness suite for H7.

    Dimensions:
    1. Alternative DVs (Roll spread, bid-ask spread)
    2. Alternative specs (Firm only, Pooled, Double-cluster)
    3. Alternative IVs (CEO-only, Presentation-only)
    4. Timing variants (concurrent, forward, lead)

    Returns:
        List of robustness result dicts
    """
    results = []

    # Dimension 1: Alternative DVs
    for dv_name, dv_label in ROBUSTNESS_CONFIG['alternative_dvs'].items():
        if dv_name not in df.columns:
            continue
        dv_fwd = f'{dv_name}__fwd'
        if dv_fwd not in df.columns:
            continue
        for uv in uncertainty_measures:
            result = run_single_regression(
                df, uv, dv_fwd, control_vars, spec='primary',
                dv_type=dv_label
            )
            if result:
                results.append(result)

    # Dimension 2: Alternative specs (use primary DV)
    for spec_name, spec_config in ROBUSTNESS_CONFIG['alternative_specs'].items():
        for uv in uncertainty_measures:
            result = run_single_regression(
                df, uv, 'illiquidity_fwd', control_vars,
                spec=spec_name, dv_type='Amihud (2002)'
            )
            if result:
                results.append(result)

    # Dimension 3: Alternative IVs
    for iv_type, ivs in ROBUSTNESS_CONFIG['alternative_ivs'].items():
        for uv in ivs:
            if uv in df.columns:
                result = run_single_regression(
                    df, uv, 'illiquidity_fwd', control_vars,
                    spec='primary', dv_type='Amihud (2002)',
                    iv_type=iv_type
                )
                if result:
                    results.append(result)

    # Dimension 4: Timing variants
    timing_variants = create_timing_variants(df, 'amihud_illiquidity', [0, -1, -2])
    for timing_name, (df_t, dv_col) in timing_variants.items():
        for uv in uncertainty_measures:
            result = run_single_regression(
                df_t, uv, dv_col, control_vars,
                spec='primary', dv_type='Amihud (2002)',
                timing=timing_name
            )
            if result:
                results.append(result)

    return results
```

def generate_results_report(primary_results, robustness_results, sample_stats):
    """
    Generate comprehensive results report.

    Includes:
    - Executive summary
    - Primary specification results
    - Robustness suite results
    - Sample statistics
    - Interpretation and conclusions
    """
    # ... existing primary results section ...

    # Add robustness section
    report += "\n## Robustness Suite Results\n\n"

    # Alternative DVs
    report += "### Alternative Dependent Variables\n\n"
    alt_dv_results = [r for r in robustness_results if r.get('dv_type') != 'Amihud (2002)']
    if alt_dv_results:
        report += "| DV | UV | Beta | SE | p | Significant? |\n"
        report += "|----|----|-----|----|----|--------------|\n"
        for r in alt_dv_results:
            sig = "Yes" if r['p_one_sided'] < 0.05 else "No"
            report += f"| {r['dv_type']} | {r['uncertainty_var']} | {r['coefficient']:.4f} | {r['se']:.4f} | {r['p_one_sided']:.4f} | {sig} |\n"
    else:
        report += "No alternative DVs available in data.\n"

    # Alternative specifications
    report += "\n### Alternative Specifications\n\n"
    for spec in ['firm_only', 'pooled', 'double_cluster']:
        spec_results = [r for r in robustness_results if r['spec'] == spec]
        n_sig = sum(1 for r in spec_results if r['p_one_sided'] < 0.05)
        report += f"- **{spec}**: {n_sig}/{len(spec_results)} significant\n"

    # Alternative IVs
    report += "\n### Alternative Independent Variables\n\n"
    for iv_type in ['ceo_only', 'presentation_only', 'qa_only']:
        iv_results = [r for r in robustness_results if r.get('iv_type') == iv_type]
        n_sig = sum(1 for r in iv_results if r['p_one_sided'] < 0.05)
        report += f"- **{iv_type}**: {n_sig}/{len(iv_results)} significant\n"

    # Timing tests
    report += "\n### Timing Tests\n\n"
    for timing in ['concurrent', 'forward', 'lead']:
        timing_results = [r for r in robustness_results if r.get('timing') == timing]
        n_sig = sum(1 for r in timing_results if r['p_one_sided'] < 0.05)
        report += f"- **{timing}**: {n_sig}/{len(timing_results)} significant\n"

    # Robustness conclusion
    report += "\n### Robustness Conclusion\n\n"
    if robustness_results:
        n_total = len([r for r in robustness_results if r['spec'] == 'primary' or r.get('timing') == 'forward'])
        n_sig = sum(1 for r in robustness_results if r['p_one_sided'] < 0.05)
        report += f"Across {n_total} robustness tests, {n_sig} ({n_sig/n_total*100:.1f}%) show significant effects.\n\n"
        if n_sig / n_total > 0.5:
            report += "**Robustness Assessment:** Results are ROBUST - majority of tests significant.\n"
        elif n_sig / n_total > 0.2:
            report += "**Robustness Assessment:** Results are MIXED - some specifications significant.\n"
        else:
            report += "**Robustness Assessment:** Results are NOT ROBUST - few specifications significant.\n"

    return report
```
