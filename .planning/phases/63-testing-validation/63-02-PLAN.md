## Use pytest.approx() for Floating Point Comparison
```python
# DON'T: Direct equality is fragile
assert result == 0.3333333333

# DO: Use pytest.approx for tolerance
assert result == pytest.approx(0.3333, rel=1e-5)  # Relative tolerance
assert result == pytest.approx(0.3333, abs=1e-6)  # Absolute tolerance
```

## Use pandas.testing.assert_frame_equal()
```python
# DON'T: No error context
assert df1.equals(df2)

# DO: Useful diagnostics
pd.testing.assert_frame_equal(df1, df2, check_dtype=False)
pd.testing.assert_frame_equal(df1, df2, rtol=1e-5)  # For floats
```

```python
@pytest.mark.parametrize("assets,expected", [

])
```

    ```python
    import pandas as pd
    import numpy as np
    from pathlib import Path
    import sys
    ```

    ```python
        from shared.iv_regression import (
            run_iv2sls,
            WeakInstrumentError,
            LINEARMODELS_AVAILABLE,
        )

    @pytest.mark.parametrize("n_instr,should_have_overid", [
        (1, False),  # Exactly identified
        (2, True),   # Over-identified
    ])
    def test_iv2sls_overidentification_test(n_instr, should_have_overid):
        """Test that overidentification test only runs when over-identified."""
        if not LINEARMODELS_AVAILABLE:
            pytest.skip("linearmodels not available")

        np.random.seed(42)
        n = 100
        df = pd.DataFrame({
            "y": np.random.randn(n),
            "x1": np.random.randn(n),
            "z1": np.random.randn(n),
            "z2": np.random.randn(n) if n_instr > 1 else np.random.randn(n),
        })

        # Create endogenous variable correlated with instruments
        df["endog"] = 0.5 * df["z1"] + 0.3 * df.get("z2", df["z1"]) + np.random.randn(n) * 0.1

        instruments = ["z1"] if n_instr == 1 else ["z1", "z2"]

        result = run_iv2sls(
            df=df,
            dependent="y",
            exog=["x1"],
            endog="endog",
            instruments=instruments,
            f_threshold=5.0,
            fail_on_weak=False,
        )

        # Check overid test
        overid = result["overid_test"]
        assert overid is not None
        assert overid["valid"] == should_have_overid
    ```
