    ```yaml
    - name: Coverage Summary
      run: |
        echo "## Coverage Summary" >> $GITHUB_STEP_SUMMARY
        echo "" >> $GITHUB_STEP_SUMMARY

        # Parse coverage.json for summary
        python - << 'EOF'
        import json
        import sys
        from pathlib import Path

        coverage_file = Path("coverage.json")
        if not coverage_file.exists():
            sys.exit(0)

        with open(coverage_file) as f:
            data = json.load(f)

        totals = data.get("totals", {})
        files = data.get("files", [])

        print("## Coverage Summary")
        print(f"| Metric | Coverage |")
        print(f"|--------|----------|")
        print(f"| Line Coverage | {totals.get('line_covered', 0)}/{totals.get('line_total', 0)} ({totals.get('line_percent', 0):.1f}%) |")
        print(f"| Branch Coverage | {totals.get('branch_covered', 0)}/{totals.get('branch_total', 0)} ({totals.get('branch_percent', 0):.1f}%) |")

        # Critical shared modules
        critical_files = [
            "financial_utils.py",
            "panel_ols.py",
            "iv_regression.py",
        ]

        print("\n### Critical Modules (Tier 1 - Target 90%+)")
        for file_info in files:
            filename = Path(file_info["name"]).name
            if filename in critical_files:
                summary = file_info.get("summary", {})
                line_pct = summary.get("percent_covered", 0)
                status = "OK" if line_pct >= 90 else "NEEDS ATTENTION"
                print(f"- **{filename}**: {line_pct:.1f}% [{status}]")

        EOF
    ```

    1. Create .coveragerc file:
    ```ini
    # .coveragerc - Coverage.py configuration
    # See: https://coverage.readthedocs.io/

    [run]
    source = 2_Scripts
    branch = true
    omit =
        */tests/*
        */__pycache__/*
        */ARCHIVE*
        */ARCHIVE_OLD*
        */V1*
        .tox/*
        */site-packages/*

    [report]
    # Minimum coverage required (used by coverage fail command)
    fail_under = 60

    # Precision for percentage reporting
    precision = 2

    # Skip files with no code
    skip_empty = True

    # Exclude lines from coverage calculation
    exclude_lines =
        pragma: no cover
        def __repr__
        raise AssertionError
        raise NotImplementedError
        if __name__ == .__main__.:
        if TYPE_CHECKING:
        @abstractmethod

    [html]
    directory = htmlcov
    title = F1D Thesis Coverage Report

    [xml]
    output = coverage.xml

    [json]
    output = coverage.json

    [lcov]
    output = coverage.lcov
    ```

    2. Add coverage fixture to tests/conftest.py:
    ```python
    @pytest.fixture(autouse=True)
    def coverage_settings(request):
        """
        Configure coverage settings for all tests.

        This fixture runs automatically (autouse=True) and configures
        coverage.py to measure branch coverage and generate reports.
        """
        # Only configure when running under coverage
        if "_cov" in request.config.pluginmanager:
            config = request.config.pluginmanager.get_plugin("_cov")

            if hasattr(config, "cov_source"):
                # Enable branch coverage
                config.opts.branch = True

        yield  # Let the test run
    ```
