    "data_validation",
    "path_utils",
    "symlink_utils",
    "string_matching",
    "dependency_checker",
    "dual_writer",  # Re-exports from observability_utils
]

for module in modules:
    try:
        __import__(module)
        print(f"✓ {module}")
    except ImportError as e:
        print(f"✗ {module}: {e}")
        errors.append((module, str(e)))

print()
if errors:
    print(f"=== Import Errors: {len(errors)} ===")
    for module, error in errors:
        print(f"{module}: {error}")
    sys.exit(1)
else:
    print("✓ All shared modules imported successfully")
    sys.exit(0)
