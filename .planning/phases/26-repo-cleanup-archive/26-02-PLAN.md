```python
#!/usr/bin/env python3
"""Generate archive inventory manifest."""

import json
from pathlib import Path
from datetime import datetime
import subprocess

def get_git_status(filepath):
    """Check if file is tracked by git."""
    try:
        subprocess.run(
            ["git", "ls-files", "--error-unmatch", str(filepath)],
            check=True,
            capture_output=True,
            cwd=filepath.parent.parent.parent  # Go up to repo root
        )
        return True
    except (subprocess.CalledProcessError, FileNotFoundError):
        return False

def create_archive_manifest():
    """Generate JSON manifest of archive contents."""
    manifest = {
        "archive_date": datetime.now().isoformat(),
        "archive_version": "1.0",
        "categories": {},
        "movements": []
    }

    archive_root = Path(".___archive")

    # Skip manifest.json itself and generate_manifest.py
    for category_dir in archive_root.iterdir():
        if not category_dir.is_dir() or category_dir.name in {".git", "__pycache__"}:
            continue

        category = category_dir.name
        file_count = 0

        for file in category_dir.rglob("*"):
            if file.is_file() and file.name not in {"manifest.json", "generate_manifest.py"}:
                file_count += 1
                relative_path = str(file.relative_to(archive_root))

                manifest["movements"].append({
                    "archive_path": relative_path,
                    "category": category,
                    "size_bytes": file.stat().st_size,
                    "git_tracked": get_git_status(file),
                    "modified": datetime.fromtimestamp(
                        file.stat().st_mtime
                    ).isoformat(),
                    "file_type": file.suffix or "none"
                })

        manifest["categories"][category] = file_count

    manifest["total_files"] = len(manifest["movements"])

    # Write manifest
    manifest_path = archive_root / "manifest.json"
    with open(manifest_path, "w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Manifest created: {manifest_path}")
    print(f"Total files archived: {manifest['total_files']}")
    print(f"\nBreakdown by category:")
    for cat, count in manifest["categories"].items():
        print(f"  {cat}: {count} files")

if __name__ == "__main__":
    create_archive_manifest()
```
