       ```python
       from lifelines import CoxPHFitter
       import pandas as pd
       from typing import Dict, Any

       def run_cox_ph(df: pd.DataFrame, time_col: str, event_col: str, formula: str) -> Dict[str, Any]:
           """
           Run Cox Proportional Hazards model using lifelines.

           Args:
               df: DataFrame with survival data
               time_col: Column name for survival time
               event_col: Column name for event indicator
               formula: R-style formula for covariates

           Returns:
               Dict with coefficients, confidence_intervals, summary, concordance_index
           """
           # Validate inputs
           missing_cols = [col for col in [time_col, event_col] if col not in df.columns]
           if missing_cols:
               raise ValueError(f"Missing columns: {missing_cols}")

           # Prepare data
           df_clean = df[[time_col, event_col] + [c.strip() for c in formula.split('+')[1:]]].copy()
           df_clean = df_clean.dropna()

           # Fit model
           cph = CoxPHFitter()
           cph.fit(df_clean, duration_col=time_col, event_col=event_col, formula=formula)

           # Return results
           return {
               "coefficients": cph.params_.to_dict(),
               "confidence_intervals": cph.confidence_intervals_.to_dict(),
               "summary": cph.summary.to_dict(),
               "concordance_index": cph.concordance_index_,
               "model": cph
           }
       ```
